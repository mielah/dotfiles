   throughPat(func: uid(pId)) @filter(eq(provider.hipaa_isolation_level, "through patient")){
    has_patient {
    	patient.first_name
      patient.middle_name
      patient.last_name
      patient_id: uid
      owns {
        healthcare_data.added_date
        healthcare_data.data_type
        healthcare_data.key
      }
    }
    
    generated {
    	healthcare_data.added_date
      healthcare_data.data_type

    }
  }


#query to fetch data for 'this org only' hipaa level
   thisOrg(func: has(belongs_to)) @filter(has(generated) AND eq(provider.hipaa_isolation_level, "this org")) @cascade {

      organization.name
      org_id: uid

    orgProvider: uid

    provider.first_name
      
    generated {
      healthcare_data.added_date
      healthcare_data.data_type
      healthcare_data.key
    }
  }
  

#query to fetch data for 'parent org' hipaa level
	parentOrg(func: has(belongs_to)) @filter(eq(provider.hipaa_isolation_level, "parent org")) @cascade {
